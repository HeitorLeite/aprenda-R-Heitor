Geração aleatória e gráfico

Este programa gera aleatóriamente uma quantidade de numeros definida pelo usuariio e depois faz um gráfico com eles

# Bibliotecas necessárias
                    library(ggplot2)
                    
                    # Função principal para gerar gráfico
                    gerar_grafico <- function() {
                      # Entrada do usuário
                      quantidade <- as.integer(readline(prompt = "Digite a quantidade de números a serem gerados: "))
                      limite_min <- as.integer(readline(prompt = "Digite o limite mínimo: "))
                      limite_max <- as.integer(readline(prompt = "Digite o limite máximo: "))
                      
                      # Geração de números inteiros aleatórios
                      numeros <- sample(limite_min:limite_max, quantidade, replace = TRUE)
                      print(paste("Números gerados:", paste(numeros, collapse = ", ")))
                      
                      # Data frame para o gráfico
                      dados_grafico <- data.frame(Índice = 1:quantidade, Números = numeros)
                      
                      # Criação do gráfico
                      grafico <- ggplot(dados_grafico, aes(x = Índice, y = Números)) +
                        geom_point(color = "red", size = 2) +  # Pontos vermelhos
                        geom_line(color = "blue") +  # Linhas azuis interligando os pontos
                        labs(title = "Números Aleatórios Gerados", x = "Índice", y = "Valor") +
                        theme_minimal() +
                        theme(
                          plot.title = element_text(size = 14, face = "bold", color = "white"),
                          axis.title.x = element_text(size = 12, color = "white"),
                          axis.title.y = element_text(size = 12, color = "white"),
                          axis.text.x = element_text(size = 10, color = "white"),
                          axis.text.y = element_text(size = 10, color = "white"),
                          panel.background = element_rect(fill = "black"),  # Fundo do painel preto
                          plot.background = element_rect(fill = "black")  # Fundo do gráfico preto
                        )
                      
                      # Salvar o gráfico como PNG
                      nome_arquivo <- "numeros_aleatorios.png"
                      ggsave(nome_arquivo, plot = grafico, width = 8, height = 6)
                      
                      cat("O gráfico dos números gerados foi salvo como:", nome_arquivo, "\n")
                    }
                    
                    # Função principal com loop para continuar ou encerrar
                    main <- function() {
                      repeat {
                        # Chama a função para gerar gráfico
                        gerar_grafico()
                        
                        # Pergunta ao usuário se deseja continuar ou encerrar
                        continuar <- readline(prompt = "Deseja gerar outro gráfico? (s/n): ")
                        if (tolower(continuar) != "s") {
                          cat("Programa encerrado.\n")
                          break
                        }
                      }
                    }
                    
                    # Chama a função principal
                    main()
